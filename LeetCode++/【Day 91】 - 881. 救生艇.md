class Solution:
    def numRescueBoats(self, people: List[int], limit: int) -> int:
        people.sort()
        i, j = 0, len(people) - 1
        num = 0
        while i <= j:
            if people[i] + people[j] > limit:
                j -= 1
            else:
                j -= 1
                i += 1
            num += 1
        return num
