```java
public class quickSort {

    public static void main(String[] args) {
        quickSort Sort = new quickSort();
        int[] nums = {5, 1, 1, 2, 0, 0};
//        int[] nums = {1, 2, 3, 0, 1, 0, 0, 1, 2, 3};
        Sort.quickSort(nums, 0, nums.length - 1);
        System.out.println(Arrays.toString(nums));
    }

    public void quickSort(int[] nums, int l, int r) {
        if (l < r) {
            int idx = partition(nums, l, r);
            quickSort(nums, l, idx - 1);
            quickSort(nums, idx + 1, r);
        }
    }

    public int partition(int[] nums, int l, int r) {
//        int idx = new Random().nextInt(r - l + 1) + l; // new Random().nextInt(n)随机产生 [0,n) 的int值
//        int idx = (int) (Math.random() * (r - l + 1)) + l;
        swap(nums, r, (int) (Math.random() * (r - l + 1)) + l);
        int i = l;
        for (int j = l; j < r; j++) {
            if (nums[j] < nums[r]) {
                swap(nums, i, j);
                i++;
            }
        }
        swap(nums, i, r);
        return i;
    }

    public void swap(int[] nums, int i, int j) {
        int temp = nums[i];
        nums[i] = nums[j];
        nums[j] = temp;
    }
}
```

```java
package quicksort;

import java.util.*;

/**
 * @author Orust
 * @create 2021/3/17 20:35
 */
public class quickSort {

    public static void main(String[] args) {
        quickSort Sort = new quickSort();
        int[] nums = {5, 1, 1, 2, 0, 0};
//        int[] nums = {1, 2, 3, 0, 1, 0, 0, 1, 2, 3};
        Sort.quickSort(nums, 0, nums.length - 1);
        System.out.println(Arrays.toString(nums));
    }

    public void quickSort(int[] nums, int l, int r) {
        if (l >= r) return;
        int i = l - 1, j = r + 1, x = nums[l];
        while (i < j) {
            do i++; while (nums[i] < x);
            do j--; while (nums[j] > x);
            if (i < j) {
                int temp = nums[i];
                nums[i] = nums[j];
                nums[j] = temp;
            }
        }
        quickSort(nums, l, j);
        quickSort(nums, j + 1, r);
    }
}
```

* 简洁版：[912. 排序数组](https://leetcode-cn.com/problems/sort-an-array/)

```java
class Solution {
    public int[] sortArray(int[] nums) {
        quickSort(nums, 0, nums.length - 1);
        return nums;
    }

    void quickSort(int[] nums, int l, int r) {
        if (l >= r) return;
        swap(nums, r, (int) (Math.random() * (r - l + 1) + l));
        int i = l;
        for (int j = l; j < r; j++) if (nums[j] < nums[r]) swap(nums, i++, j);

        swap(nums, i, r);
        quickSort(nums, l, i - 1);
        quickSort(nums, i + 1, r);
    }

    void swap(int[] nums, int i, int j) {
        int temp = nums[i]; nums[i] = nums[j]; nums[j] = temp;
    }
}
```
